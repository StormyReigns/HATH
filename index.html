This will return '' for uploaded PDFs. */ function extractPdfText(fileId) { try { if (typeof Drive === 'undefined') return ''; var blob = Drive.Files.get(fileId, { alt: 'media' }); // For native PDF, get() returns the raw bytes - we can't parse PDF text in Apps Script. return ''; } catch (e) { return ''; } } /** * Get or build the document index (with caching). */ function getDocIndex() { if (DOCS_CACHE && (Date.now() - DOCS_CACHE_TIME) < CACHE_TTL_MS) { return DOCS_CACHE; } DOCS_CACHE = listFilesRecursive(DOCS_FOLDER_ID); DOCS_CACHE_TIME = Date.now(); return DOCS_CACHE; } /** * Score a file's relevance to the query. * Higher = more relevant. Match filename and folder path first, then content. */ function scoreRelevance(file, query, content) { var q = query.toLowerCase(); var name = (file.name || '').toLowerCase(); var path = (file.path || '').toLowerCase(); var score = 0; var qWords = q.split(/\s+/).filter(Boolean); qWords.forEach(function(w) { if (w.length < 3) return; if (name.indexOf(w) >= 0) score += 10; if (path.indexOf(w) >= 0) score += 5; if (content && content.toLowerCase().indexOf(w) >= 0) score += 2; }); if (name.indexOf(q) >= 0 || q.indexOf(name) >= 0) score += 20; return score; } /** * Get document context for a query. * Returns { text: string, cited: [names] } with content capped at MAX_CONTEXT_WORDS. */ function getDocumentContext(query) { if (!query || typeof query !== 'string') return { text: '', cited: [] }; var index = getDocIndex(); if (index.length === 0) return { text: '', cited: [] }; var scored = []; for (var i = 0; i < index.length; i++) { var file = index[i]; var content = ''; var mime = file.mimeType || ''; if (mime === 'application/vnd.google-apps.document') { content = extractDocText(file.id); } else if (mime === 'application/pdf') { content = extractPdfText(file.id); if (!content) content = ''; // PDF: use name/path only } // Other types: content stays '' var score = scoreRelevance(file, query, content); if (score > 0) { scored.push({ file: file, content: content, score: score }); } } scored.sort(function(a, b) { return b.score - a.score; }); var totalWords = 0; var parts = []; var cited = []; for (var j = 0; j < scored.length && totalWords < MAX_CONTEXT_WORDS; j++) { var s = scored[j]; var name = s.file.name; var path = s.file.path; var text = s.content; var entry = '--- ' + name + ' (path: ' + path + ') ---\n'; if (text) { var words = text.split(/\s+/).filter(Boolean); var remaining = MAX_CONTEXT_WORDS - totalWords; var take = Math.min(words.length, remaining); entry += words.slice(0, take).join(' '); if (take < words.length) entry += ' [truncated]'; totalWords += take; } else { entry += '[No text extracted - file name and path only]'; } parts.push(entry); cited.push(name); } return { text: parts.length > 0 ? 'DOCUMENTS:\n\n' + parts.join('\n\n') : '', cited: cited }; } /** * List contents of a folder (non-recursive) for the Documents screen. * @param {string} folderId - Drive folder ID * @returns {Object} { folders: [{id, name}], files: [{id, name, mimeType, url}], folderName } */ function listFolder(folderId) { try { var folder = DriveApp.getFolderById(folderId); var folderName = folder.getName(); var folders = []; var files = []; var subfolders = folder.getFolders(); while (subfolders.hasNext()) { var f = subfolders.next(); folders.push({ id: f.getId(), name: f.getName() }); } folders.sort(function(a, b) { return (a.name || '').localeCompare(b.name || ''); }); var fileIter = folder.getFiles(); while (fileIter.hasNext()) { var file = fileIter.next(); files.push({ id: file.getId(), name: file.getName(), mimeType: file.getMimeType() || '', url: file.getUrl() }); } files.sort(function(a, b) { return (a.name || '').localeCompare(b.name || ''); }); return { folders: folders, files: files, folderName: folderName }; } catch (e) { Logger.log('listFolder error: ' + e); return { folders: [], files: [], folderName: '', error: String(e) }; } } /** * INTEGRATION: Add to doGet(e): * var params = e.parameter || {}; * if (params.action === 'listFolder' && params.folder) { * var data = listFolder(params.folder); * return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); * } * * INTEGRATION: In your doPost, after getting the user message and before calling Claude: * * 1. var docContext = getDocumentContext(parsed.rawMessage); * * 2.
